cmake_minimum_required(VERSION 3.7.2)
project(libtest_pkg)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  rospy
  std_msgs
  actionlib
  actionlib_msgs
  message_generation
)

## System dependencies are found with CMake's conventions
find_package(Boost REQUIRED COMPONENTS system)

## Generate actions in the 'action' folder
 add_action_files(
   FILES
   Servo.action
 )

## Generate added messages and services with any dependencies listed here
 generate_messages(
   DEPENDENCIES
   std_msgs
   actionlib_msgs
 )

catkin_package(
  CATKIN_DEPENDS roscpp rospy std_msgs actionlib actionlib_msgs message_runtime 
)

include_directories(
  include
  ${catkin_INCLUDE_DIRS}
)


add_library(SimpleServo src/SimpleServo.cpp)

add_executable(servo_actionserver src/servo_actionserver.cpp)
add_dependencies(servo_actionserver libtest_pkg_generate_messages_cpp)
target_link_libraries(servo_actionserver ${catkin_LIBRARIES} SimpleServo)

add_executable(servo_actionclient src/servo_actionclient.cpp)
add_dependencies(servo_actionclient libtest_pkg_generate_messages_cpp)
target_link_libraries(servo_actionclient ${catkin_LIBRARIES})
